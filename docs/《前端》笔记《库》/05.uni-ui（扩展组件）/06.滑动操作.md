## 一、基础结构
  ```html
  <uni-swipe-action>
    <uni-swipe-action-item :right-options="options" @click="slidingXXX">
      <!-- 把内容放到 item 盒子中，不然布局容易出问题 -->
      <view class="item">
      </view>
    </uni-swipe-action-item>
  </uni-swipe-action>
  ```

  ```js
  data() {
    return {
      options: [{
        text: '删除', // 文本内容
        style: {
          backgroundColor: '#C00000' // 按钮的背景颜色
        }
      }]
    }
  },
  methods: {
    sliding() { // 滑动XXX

    }
  }
  ```
