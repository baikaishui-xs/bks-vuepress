# 配置项
  ```js
  var dashBoardChartConfig: { // 仪表盘 配置项
    title: { // 标题组件
      text: '仪表盘' // 主标题
    },
    series: [
      {
        type: 'gauge',  // 图表类型：仪表盘
        data: [
          { // 每一个对象就代表一个指针
            value: 97,
            itemStyle: {  // 指针的样式
              color: 'pink'  // 指针的颜色
            }
          },
          {
            value: 85,
            itemStyle: {
              color: 'green'
            }
          }
        ],
        min: 50,  // 仪表盘数值范围
        max: 100
      }
    ]
  }
  ```
