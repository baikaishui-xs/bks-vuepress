Code → 首选项 → 用户片段 → 选择使用自定义片段的文件格式 → 使用以下模板进行自定义片段

// 创建代码块
"Print to console": {  // 代码块名称

		// 代码块提示语
		"prefix": "!!",
			
		// 代码块内容
				// $1~100 代表代码块出现后的光标位置
				// $1 $1 $1 可以有多个相同的，代表同时编辑多个地方
"body": [
		"<!DOCTYPE html>;",
		"<html lang=\"en\">",
		"<head>",
		"    <meta charset=\"UTF-8\">",
		"    <title>Document</title>",
		"</head>",
		"<body>",
		"    $1",
		"</body>",
		"</html>"
			],
			
		// 代码块描述
"description": "html 骨架"
}
